# Core pipeline — everything the default command needs:
#     voxa video.mp4     (whisper transcription + google translation + edge TTS)
#
# Optional engines live in extras, so you only install what you use:
#     pip install "voxa[xtts]"        voice cloning  ⚠️ non-commercial weights, see NOTICE.md
#     pip install "voxa[piper]"       fully offline TTS
#     pip install "voxa[faster]"      faster-whisper backend (2-4x, no torch)
#     pip install "voxa[anthropic]"   Claude translation

# ── Core ─────────────────────────────────────────────────
openai-whisper          # transcription (installs torch as a dependency)
edge-tts>=7.0           # default TTS engine (older versions hit a Microsoft 403)
deep-translator         # Google translation backend
openai                  # OpenAI translation + OpenAI TTS
tqdm                    # progress bars
requests                # Ollama HTTP calls
torch                   # inference runtime

# ── Audio processing ─────────────────────────────────────
soundfile               # WAV read/write
noisereduce             # denoising of the *source* audio before transcription
numpy
